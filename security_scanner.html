#!/usr/bin/env python3
# WebShield Scanner — Sadece kendi sitenizi test edin
# pip install requests beautifulsoup4 colorama urllib3
# python scanner.py --url https://siteniz.com --all

import argparse, requests, re, json, sys
from urllib.parse import urlparse, urljoin, parse_qs
from bs4 import BeautifulSoup
from colorama import Fore, Style, init
import urllib3
urllib3.disable_warnings()
init(autoreset=True)

# ─── Renkli çıktı ────────────────────────────────────────────────────────────
def ok(msg):   print(Fore.GREEN  + "[✓] " + msg)
def warn(msg): print(Fore.YELLOW + "[!] " + msg)
def err(msg):  print(Fore.RED    + "[✗] " + msg)
def info(msg): print(Fore.CYAN   + "[i] " + msg)
def hdr(msg):  print("\n" + Style.BRIGHT + "═"*55 + "\n  " + msg + "\n" + "═"*55)

findings = []

def add_finding(title, severity, detail, fix):
    findings.append({"title": title, "severity": severity,
                     "detail": detail, "fix": fix})

# ─── HTTP oturumu ─────────────────────────────────────────────────────────────
def make_session():
    s = requests.Session()
    s.headers.update({"User-Agent": "WebShield-Scanner/1.0 (authorized-security-test)"})
    s.verify = False
    return s

# ─── Form toplayıcı ───────────────────────────────────────────────────────────
def collect_forms(url, session):
    try:
        r = session.get(url, timeout=10)
        soup = BeautifulSoup(r.text, "html.parser")
        forms = []
        for form in soup.find_all("form"):
            action = form.get("action", "")
            method = form.get("method", "get").lower()
            inputs = {}
            for inp in form.find_all(["input", "textarea", "select"]):
                name = inp.get("name")
                if name:
                    inputs[name] = inp.get("value", "test")
            forms.append({"action": urljoin(url, action),
                          "method": method, "inputs": inputs})
        return forms, soup
    except Exception as e:
        warn(f"Form toplama hatası: {e}")
        return [], None

# ─── 1. SQL Injection ─────────────────────────────────────────────────────────
SQL_PAYLOADS = [
    "'", '"', "' OR '1'='1", "' OR 1=1--",
    "' AND SLEEP(3)--", "' UNION SELECT NULL--",
    "admin'--", "1; DROP TABLE users--",
]
SQL_ERRORS = [
    "sql syntax", "mysql_fetch", "ORA-", "syntax error",
    "unclosed quotation", "pg_query", "sqlite3", "SQLSTATE",
    "Warning: mysql", "PDOException",
]

def test_sqli(url, session):
    hdr("SQL Injection Testi")
    forms, _ = collect_forms(url, session)
    found = False
    parsed = urlparse(url)
    params = parse_qs(parsed.query)
    if params:
        for param in params:
            for payload in SQL_PAYLOADS:
                test_params = {k: (payload if k == param else v[0])
                               for k, v in params.items()}
                try:
                    r = session.get(url.split("?")[0],
                                    params=test_params, timeout=8)
                    if any(e.lower() in r.text.lower() for e in SQL_ERRORS):
                        err(f"SQLi bulundu! Param: {param} | Payload: {payload}")
                        add_finding("SQL Injection (URL)", "HIGH",
                            f"'{param}' parametresi SQL hatasına neden oldu.",
                            "Parametrized query / prepared statement kullanın.")
                        found = True
                        break
                except: pass
    if not found: ok("SQL Injection açığı tespit edilmedi.")
